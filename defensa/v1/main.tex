	\begin{frame}[fragile]{Problema de la palabra.}
		Sea $ G = \langle X \mid R \rangle$ grupo finitamente presentado.
		
		\pause 

		El \blue{problema de la palabra} consiste en:
		\pause 
		\begin{itemize}
					\item 
						\textbf{Entrada}: Una palabra $w$ en $X \cup X^{-1}$.
					\pause 
					\item 
						\textbf{Pregunta}: ¿Vale $w=1$ en $G$?
		\end{itemize}
		\pause 

		El problema de la palabra no es \alert{decidible}.
		\pause 
		\begin{itemize}
			\item 
				Existen grupos que no admiten un algoritmo que decida si una palabra representa la identidad o no.
		\end{itemize}
		
		
	\end{frame}

\begin{frame}[fragile]{Grupos libres.}
	Dado un conjunto $X$ notamos $F_{X}$ al \emph{grupo libre} con base $X$.

	% \onslide<2->
	% El grupo $F_{X}$ viene con una función $\iota: X \to F_{X}$ que denominamos la inclusión de los generadores en el grupo libre y queda caracterizado por la siguiente propiedad universal: 
	% \onslide<3->
	% Para todo grupo $H$ y toda función $f:X \to H$ existe un único morfismo de grupos $\ol f: F_{X} \to H$ tal que $\ol f \circ \iota = f$.
	% \begin{center}
	% 	\begin{tikzcd}
	% 		F_{X}  \arrow[rr, "\ol f", dashed]          &  & H \\
	% 		&  &   \\
	% 		X \arrow[uu, "\iota"] \arrow[rruu, "f", swap] &  &  
	% 	\end{tikzcd}
	% \end{center}


	\onslide<+->
	Identificamos los elementos de $F_{X}$ con las palabras reducidas en $X \cup X^{-1}$.

	\defcajita{Un grupo $G$ es \blue{virtualmente libre} si es finitamente generado y
		tiene un subgrupo libre $F$ tal que $[G:F] < \infty$.}
		\textbf{Ejemplos}.
			\begin{itemize}
				\item \pause Los grupos finitos.
				\item \pause Los grupos libres.
				\item \pause El producto semidirecto de un grupo libre con un grupo finito.
				\item \pause El producto libre de dos grupos finitos.
			\end{itemize}
\end{frame}

		\defcajita{Un grupo $G$ es \blue{virtualmente libre} si es finitamente generado y
		tiene un subgrupo libre $F$ tal que $[G:F] < \infty$.}
		\onslide<2->
		\textbf{Ejemplos}.
			\begin{itemize}

				\item \pause Los grupos libres.

	\end{frame}

	\begin{frame}[fragile]{Teoría de lenguajes.}
		Dado un conjunto finito $\Sigma$ notaremos por $\Sigma^*$ al monoide libre sobre $\Sigma$.
		\pause 
		Un \blue{lenguaje} $L$ sobre un alfabeto $\Sigma$ es un subconjunto de $\Sigma^*$.
		\pause 	

			\textbf{Ejemplos}.


			\begin{itemize}
				\item 
					Dado $\Sigma = \{a,b\}$ consideramos el lenguaje de los palíndromos
					\[
						L = \{ w \in \Sigma^{*} \mid w = w^{R}  \}.
					\]
				\item \pause 
					Dado un grupo $G$ finitamente generado por $X$ consideramos el lenguaje
					\[
						\WP{G}{X} = \{ w \in (X \cup X^{-1})^{*} \mid w = 1 \ \text{en $G$} \}.	
					\]
			\end{itemize}
	\end{frame}
	
	\begin{frame}[fragile]{Ejemplo de gramática.}
		Consideremos las variables $V =\{ S,A \}$ y el alfabeto $\Sigma = \{ a,b \}$.
		\pause 
		Las siguientes reglas determinan una gramática:
		\begin{align*}
			S  & \to aAS  \\
			S  & \to b    \\
			A  & \to baAb \\
			aA & \to bbb
		\end{align*}

		\pause 

		A partir de una gramática podemos derivar palabras.

		\pause 
		\textbf{Ejemplo}: podemos derivar la palabra $abbbbbb$ de la siguiente manera.  
		\[
			S \to aAS \to abaAbS \to abaAbb \to abbbbbb	
		\]
	\end{frame}

	\begin{frame}[fragile]{Gramáticas (pt.1).}
		\tcajita{Definición}{Una \blue{gramática} es una tupla ${\cal G} = (V, \Sigma, P, S)$ donde:
		\pause 
		\begin{itemize}
			\item $V$ es un conjunto finito denominado las \blue{variables};
			\pause 
			\item $\Sigma$ es un conjunto finito disjunto de $V$ que denominamos el \blue{alfabeto};
			\pause 
			\item $P \subseteq ((V \cup \Sigma)^{*} - \Sigma^{*}) \times (V \cup \Sigma)^*$ es un conjunto finito de \blue{producciones}; 
			\pause 
			\item $S \in V$ es el \blue{símbolo inicial}.
			\pause 
		\end{itemize}}
		\onslide<6->{Es usual escribir las producciones $(\gamma, \nu) \in P$ como $\gamma \to \nu$.}
	\end{frame}
	
	\begin{frame}[fragile]{Gramáticas (pt.2).}
		Podemos considerar la relación sobre 
		$(V \cup \Sigma)^{*}$ dada por 
		\[
		\delta \ \gamma \ \eta \to \delta \  \nu \ \eta 
		\]
		si $\gamma \to \nu \in P$ y $\delta, \eta \in (V \cup \Sigma)^{*}$.
		
		\pause 


		Denotaremos por $\tto$ la clausura reflexiva y transitiva de $\to$.
		\pause 
		Si $\gamma \tto \delta$ diremos que $\gamma$ \blue{deriva} en $\delta$.
		\pause 
		\tcajita{Definición.}{Dada una gramática $\cGg$  definimos el \blue{lenguaje generado por la gramática} como
		\[
		L({\cal G}) = \{ w \in \Sigma^* \ | \ S \tto w   \}.
		\]}

	\end{frame}


	\begin{frame}[fragile]{Gramáticas independiente de contexto.}
		\defcajita{Una gramática $\gramatica $ es \blue{independiente de contexto} si las producciones tienen la siguiente forma:
		\[
			A \to w
		\]
		donde $A \in V, w \in (\Sigma \cup V)^*$.}
		\pause 
		\bigskip 

		% \defcajita{Si $L=\lengderivado$ para alguna gramática independiente de contexto $\cal G$ entonces diremos que $L$ es un \blue{lenguaje independiente de contexto}.}
		\defcajita{Un lenguaje $L$ es \blue{ independiente de contexto} si existe $\cG$ gramática \ic{} tal que 
		$L = \lengderivado$.}

	\end{frame}

	\begin{frame}{Descomposición en un árbol y treewidth de un grafo.}
	Sea $\Gamma$ un grafo no dirigido.
	Una \blue{descomposición en un árbol} es un par $(T,f)$ donde
	$T$ es un árbol y $f$ una función 
	\[
	f: V(T) \to \partes{V(\Gamma)}
	\]
	que cumple las siguientes condiciones:
	\pause 
	\begin{enumerate}
		\item Para todo vértice $v \in V(\Gamma)$ el subgrafo inducido por $\{ t \in V(T) \mid v \in f(t)\}$ es un árbol.

		\item Para toda arista $\{v,w\} \in E(\Gamma)$ 
		 existe $t \in V(T)$ tal que $v,w \in f(t)$. \pause 
		
	\end{enumerate}
	\end{frame}

	\begin{frame}[fragile]{Treewidth.}
		\defcajita{Sea $\Gamma$ un grafo y $(T,f)$ una descomposición en un árbol.
			La \blue{anchura} de la descomposición es el siguiente valor:
			\begin{equation*}
				a(T,f) = \sup_{t \in V(T)} |f(t)| - 1 \in \  \NN \cup \{ +\infty \}.
			\end{equation*}}
		\pause 

		\defcajita{El \blue{treewidth} de un grafo $\Gamma$ es 
		la mínima anchura posible de una descomposición en un árbol.
		% \begin{equation*}
			% tw(\Gamma) = \min \{  a(T,f)  \mid (T,f) \ \text{descomposición en un árbol} \}
		% \end{equation*}
		}

	\end{frame}

	\begin{frame}[fragile]{Grafos como espacios métricos.}
		Sea $\Gamma$ un grafo conexo luego podemos considerar el espacio métrico dado por $(V(\Gamma), d)$ donde:
		\[
			d(x,y) = \min \{ |\gamma| : \gamma \ \text{camino de} \ x  \ \text{a} \ y  \}	
		\]
		\pause 
		si $\gamma$ es un camino que realiza la distancia entre $x$ e $y$ diremos que $\gamma$ es una \blue{geodésica}.
		\pause 
		\defcajita{Sea $\Gamma$ un grafo conexo luego dado un conjunto $C$ definimos el \blue{borde de vértices} de $C$ como
		\[
			\beta(C) = B_{1}\left[C\right] \cap B_{1}\left[C^{c}\right]	
		\]}
	\end{frame}

	\begin{frame}[fragile]{Descomponiendo el grafo de Cayley en un árbol.}
		Sea $G$ grupo finitamente generado por $X$ y sea $\Gamma$ su grafo de Cayley.
		
		\pause 
		Para cada $n \in \NN_{0}$ consideramos:
		\[
			V_{n} = \{C : C \ \text{es componente conexa de} \ \Gamma \setminus B_{n}[1]\}	
		\]
		\pause 
		Consideramos el siguiente árbol $T$ con raíz $\{1\}$:
		\pause 
		\[
			V(T) = \{ \{1\}\} \cup \bigcup_{n \in \NN_{0}}  \{ \beta C : C \in V_{n} \}
		\]
		\pause 
		\[
			E(T) = \{ \{\beta C, \beta D\} : C \in V_{n}, D \in V_{n+1}, D \subseteq C\} \cup 
			\{ \{1, \beta C\} : C \in V_{0} \}
		\]
		\pause 
		
		Definimos $f: V(T) \to \partes{V(\Gamma)}$ por medio de $f(Y) = Y$ para todo $Y \in V(T)$.
		\pause 
		
		\lemacajita{$(T,f)$ es una descomposición en un árbol.}
		
	\end{frame}
